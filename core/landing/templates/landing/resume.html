CREATE 5S FADE IN
(*Tested in QLab 3.2.14. Original from Rich Walsh template.*)

--not setting level of fade when target is a fade, must do manually, other q types work.


tell application id "com.figure53.qlab.3" to tell front «class qDoc»


set userDuration to 5
set userBump to 5
set originalCue to last item of («class qSEL» as list)
set originalCueType to «class qTyp» of originalCue
set qnameStr to "Fade out: " & «class qNam» of originalCue

set «class qCon» of originalCue to «constant ContDoCo»
originalCue «event QLablevs» given «class levR»:0, «class levC»:0, «class levD»:-70
if originalCueType is in {"Group", "Audio", "Video"} then
«event QLabmake» given «class newT»:"Fade"
set newCue to last item of («class qSEL» as list)
set properties of newCue to {«class qQTr»:originalCue, «class qDur»:userDuration, «class qNam»:qnameStr}
newCue «event QLablevs» given «class levR»:0, «class levC»:0, «class levD»:0

if originalCueType is "Video" then
set «class qOpa» of newCue to 100
set «class qDoO» of newCue to true
set «class qOpa» of originalCue to 0
set «class qDoO» of originalCue to true

end if

set «class qNam» of newCue to "Fade up: " & «class qNam» of originalCue


else if originalCueType is "Fade" then
set originalCueTarget to «class qQTr» of originalCue
«event QLabmake» given «class newT»:"Fade"
set newCue to last item of («class qSEL» as list)
set properties of newCue to {«class qQTr»:originalCue, «class qDur»:userDuration, «class qNam»:qnameStr}

(*
set currentLevel to newCue getLevel row 0 column 0
set newLevel to currentLevel + userBump
newCue setLevel row 0 column 0 "+" & newLevel
*)

(*
if q type of originalCueTarget is not "Video" then
set stop target when done of newCue to true
end if
*)

set «class qNam» of newCue to "Fade up: " & «class qNam» of originalCueTarget
end if
end tell
